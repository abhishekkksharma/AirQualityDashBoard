<script>
  const firebaseConfig = {
    apiKey: "AIzaSyAnOlSp29VeB0AIaeKSk9G-Y3R3f70XrGU",
    databaseURL: "https://iot-project-2fce4-default-rtdb.firebaseio.com/"
  };

  firebase.initializeApp(firebaseConfig);
  const db = firebase.database();

  // ── Live sensor cards ──────────────────────────────────────────
  db.ref("sensor").on("value", (snapshot) => {
    const data = snapshot.val();
    if (!data) return;

    document.getElementById("temp").innerText     = data.temperature + "°";
    document.getElementById("humidity").innerText = data.humidity   ?? "--";
    document.getElementById("mq135").innerText    = data.mq135      ?? "--";
    document.getElementById("co2").innerText      = data.co2        ?? "N/A";

    const badge = document.getElementById("status-badge");
    let label = "Good", cls = "badge-good";
    if (data.mq135 > 3000)      { label = "Poor";     cls = "badge-poor"; }
    else if (data.mq135 > 2000) { label = "Moderate"; cls = "badge-moderate"; }
    badge.textContent = label;
    badge.className   = "status-badge " + cls;

    const now = new Date();
    document.getElementById("last-updated").innerText =
      "Updated " + now.toLocaleTimeString([], { hour:'2-digit', minute:'2-digit', second:'2-digit' });
  });

  // ── History table ──────────────────────────────────────────────
  const MAX_ROWS = 20;

  db.ref("readings").orderByKey().limitToLast(MAX_ROWS).on("value", (snapshot) => {
    const data  = snapshot.val();
    const tbody = document.getElementById("history-body");

    if (!data) {
      tbody.innerHTML = `<tr><td colspan="5" class="history-empty">No history yet</td></tr>`;
      document.getElementById("count-badge").textContent = "0 entries";
      return;
    }

    const rows = Object.values(data).reverse(); // newest first
    document.getElementById("count-badge").textContent = rows.length + " entr" + (rows.length === 1 ? "y" : "ies");

    tbody.innerHTML = rows.map(r => {
      const time = r.ts
        ? new Date(r.ts * 1000).toLocaleTimeString([], { hour:'2-digit', minute:'2-digit', second:'2-digit' })
        : "--:--";
      const temp = r.temperature != null ? r.temperature.toFixed(1) + "°"  : "--";
      const hum  = r.humidity    != null ? r.humidity.toFixed(1)    + "%"   : "--";
      const mq   = r.mq135       != null ? r.mq135                          : "--";
      const co2  = r.co2         != null ? r.co2 + " ppm"                   : "N/A";

      return `
        <tr>
          <td class="col-time">${time}</td>
          <td class="col-temp" style="text-align:right">${temp}</td>
          <td class="col-hum"  style="text-align:right">${hum}</td>
          <td class="col-mq"   style="text-align:right">${mq}</td>
          <td class="col-co2"  style="text-align:right">${co2}</td>
        </tr>`;
    }).join("");
  });

  function clearHistory() {
    if (confirm("Delete all history from Firebase? This cannot be undone.")) {
      db.ref("readings").remove();
    }
  }
</script>
